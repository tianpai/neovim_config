# neovim_config

1. `sh -c 'curl -fLo "${XDG_DATA_HOME:-$HOME/.local/share}"/nvim/site/autoload/plug.vim --create-dirs \
       https://raw.githubusercontent.com/junegunn/vim-plug/master/plug.vim'`
2. `mkdir  ~/.config/nvim`
3. `cp __this init.vim__ $HOME/.config/nvim/`
